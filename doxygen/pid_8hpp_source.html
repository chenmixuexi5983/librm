51template <PIDType type>
52class PID {
53 public:
54 PID() = default;
55 virtual ~PID() = default;
56
57 PID(f32 kp, f32 ki, f32 kd, f32 max_out, f32 max_iout)
58 : kp_(kp), ki_(ki), kd_(kd), max_out_(max_out), max_iout_(max_iout) {}
59 virtual void Update(f32 error);
60 virtual void UpdateExtDiff(f32 error, f32 external_diff);
61 void Update(f32 set, f32 ref) { Update(set - ref); }
62 void UpdateExtDiff(f32 set, f32 ref, f32 external_diff) { Update(set - ref, external_diff); }
63 void Clear() {
64 set_ = 0;
65 fdb_ = 0;
66 out_ = 0;
67 p_out_ = 0;
68 i_out_ = 0;
69 d_out_ = 0;
70 std::memset(d_buf_, 0, sizeof(d_buf_));
71 std::memset(error_, 0, sizeof(error_));
72 }
73 [[nodiscard]] f32 value() const { return utils::absConstrain(p_out_ + i_out_ + d_out_, max_out_); }
74 [[nodiscard]] f32 p_out() const { return p_out_; }
75 [[nodiscard]] f32 i_out() const { return i_out_; }
76 [[nodiscard]] f32 d_out() const { return d_out_; }
77
78 public:
79 f32 kp_{};
80 f32 ki_{};
81 f32 kd_{};
82
83 f32 max_out_{};
84 f32 max_iout_{};
85
86 protected:
87 f32 set_{};
88 f32 fdb_{};
89
90 f32 out_{};
91 f32 p_out_{};
92 f32 i_out_{};
93 f32 d_out_{};
94 f32 d_buf_[3]{}; // 0: 这次, 1: 上次, 2: 上上次
95 f32 error_[3]{}; // 0: 这次, 1: 上次, 2: 上上次
96};

98template <>
99inline void PID<PIDType::kPosition>::Update(f32 error) {
100 error_[2] = error_[1];
101 error_[1] = error_[0];
102 error_[0] = error;
103
104 p_out_ = kp_ * error_[0];
105 i_out_ += ki_ * error_[0];
106
107 // update derivative term
108 d_buf_[2] = d_buf_[1];
109 d_buf_[1] = d_buf_[0];
110 d_buf_[0] = error_[0] - error_[1];
111
112 d_out_ = kd_ * d_buf_[0];
113 i_out_ = utils::absConstrain(i_out_, max_iout_);
114}

116template <>

121
122 p_out_ = kp_ * (error_[0] - error_[1]);

129 d_out_ = kd_ * d_buf_[0];
130}
