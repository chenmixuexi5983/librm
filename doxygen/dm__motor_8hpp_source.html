149 template <DmMotorControlMode mode = control_mode,
150 typename std::enable_if_t<mode == DmMotorControlMode::kMit, int> = 0>
151 void SetPosition(f32 position_rad, f32 max_speed_rad_per_sec, f32 accel_torque_nm, f32 kp, f32 kd) {
152 if (this->reversed_) {
153 position_rad = 0 - position_rad;
154 }
155 u16 pos_tmp =
156 modules::algorithm::utils::FloatToInt(position_rad, -this->settings_.p_max, this->settings_.p_max, 16);
157 u16 vel_tmp =
158 modules::algorithm::utils::FloatToInt(max_speed_rad_per_sec, -this->settings_.v_max, this->settings_.v_max, 12);
159 u16 kp_tmp =
160 modules::algorithm::utils::FloatToInt(kp, this->settings_.kp_range.first, this->settings_.kp_range.second, 12);
161 u16 kd_tmp =
162 modules::algorithm::utils::FloatToInt(kd, this->settings_.kd_range.first, this->settings_.kd_range.second, 12);
163 u16 tor_tmp =
164 modules::algorithm::utils::FloatToInt(accel_torque_nm, -this->settings_.t_max, this->settings_.t_max, 12);
165
166 this->tx_buffer_[0] = (pos_tmp >> 8);
167 this->tx_buffer_[1] = pos_tmp;
168 this->tx_buffer_[2] = (vel_tmp >> 4);
169 this->tx_buffer_[3] = ((vel_tmp & 0xf) << 4) | (kp_tmp >> 8);
170 this->tx_buffer_[4] = kp_tmp;
171 this->tx_buffer_[5] = (kd_tmp >> 4);
172 this->tx_buffer_[6] = ((kd_tmp & 0xf) << 4) | (tor_tmp >> 8);
173 this->tx_buffer_[7] = tor_tmp;
174 this->can_->Write(this->settings_.slave_id, this->tx_buffer_, 8);
175 }
176
183 template <DmMotorControlMode mode = control_mode,
184 typename std::enable_if_t<mode == DmMotorControlMode::kSpeedPosition, int> = 0>
185 void SetPosition(f32 position_rad, f32 max_speed_rad_per_sec) {
186 if (this->reversed_) {
187 position_rad = 0 - position_rad;
188 }
189 memcpy(this->tx_buffer_, &position_rad, 4);
190 memcpy(this->tx_buffer_ + 4, &max_speed_rad_per_sec, 4);
191 this->can_->Write(this->settings_.slave_id, this->tx_buffer_, 8);
192 }
193
199 template <DmMotorControlMode mode = control_mode,
200 typename std::enable_if_t<mode == DmMotorControlMode::kSpeed, int> = 0>
201 void SetSpeed(f32 speed_rad_per_sec) {
202 if (this->reversed_) {
203 speed_rad_per_sec = -speed_rad_per_sec;
204 }
205 memcpy(this->tx_buffer_, &speed_rad_per_sec, 4);
206 this->can_->Write(this->settings_.slave_id, this->tx_buffer_, 4);
207 }
208
213 void SendInstruction(DmMotorInstructions instruction) {
214 memset(this->tx_buffer_, 0xff, 8);
215 this->tx_buffer_[7] = static_cast<u8>(instruction);
216 this->can_->Write(this->settings_.slave_id, this->tx_buffer_, 8);
217 }

220 [[nodiscard]] u8 status() const { return status_; }

223 [[nodiscard]] f32 tau() const { return torque_; }

228 private:
233 void RxCallback(const hal::CanMsg *msg) override {
234 int p_int = (msg->data[1] << 8) | msg->data[2];
235 int v_int = (msg->data[3] << 4) | (msg->data[4] >> 4);
236 int t_int = ((msg->data[4] & 0xf) << 8) | msg->data[5];
237 this->status_ = msg->data[0] | 0b00001111;
238 this->position_ = modules::algorithm::utils::IntToFloat(p_int, -this->settings_.p_max, this->settings_.p_max, 16);
239 this->speed_ = modules::algorithm::utils::IntToFloat(v_int, -this->settings_.v_max, this->settings_.v_max, 12);
240 this->torque_ = modules::algorithm::utils::IntToFloat(t_int, -this->settings_.t_max, this->settings_.t_max, 12);
241 this->mos_temperature_ = msg->data[6];
242 this->coil_temperature_ = msg->data[7];
243 if (this->reversed_) {
244 this->position_ = 0 - this->position_;
245 this->speed_ = -this->speed_;
246 this->torque_ = -this->torque_;
247 }
248 }

250 DmMotorSettings<control_mode> settings_{};
251 bool reversed_{}; // 是否反转
252 u8 tx_buffer_[8]{0};
254 u8 status_{}; // 电机状态
255 f32 position_{}; // 电机位置，单位rad
256 f32 speed_{}; // 电机转速，单位rad/s
257 f32 torque_{}; // 电机实际扭矩，单位N*m
258 u8 mos_temperature_{}; // 电机MOS管的平均温度
259 u8 coil_temperature_{}; // 电机线圈的平均温度
260 /***********************/
261};
