48inline void SleepMs(u32 ms) {
49#ifdef LIBRM_USE_CMSIS_RTOS
50 if (__get_IPSR()) { // 检测当前是否在中断里，如果在中断里则调用HAL_Delay，否则调用osDelay

52 } else {

64inline void SleepUs(u32 us) {
65 if (!(CoreDebug->DEMCR & CoreDebug_DEMCR_TRCENA_Msk)) {
66 CoreDebug->DEMCR |= CoreDebug_DEMCR_TRCENA_Msk;
67 }
68 if (!(DWT->CTRL & DWT_CTRL_CYCCNTENA_Msk)) {
69 DWT->CTRL |= DWT_CTRL_CYCCNTENA_Msk;
70 }
71
72 u32 start = DWT->CYCCNT;
73 u32 ticks = us * (HAL_RCC_GetSysClockFreq() / 1000000);
74
75 while ((DWT->CYCCNT - start) < ticks) {
76 ;
77 }
78}
79#endif

84template <typename Rep, typename Period>
85void Sleep(const std::chrono::duration<Rep, Period> &duration) {
86#if defined(LIBRM_PLATFORM_STM32)
87 if (duration < std::chrono::milliseconds(1)) {
88 SleepUs(std::chrono::duration_cast<std::chrono::microseconds>(duration).count());
89 } else {
90 SleepMs(std::chrono::duration_cast<std::chrono::milliseconds>(duration).count());
91 }
92#else
93 std::this_thread::sleep_for(duration);
94#endif
95}
