44template <RefereeRevision revision>
45class Referee {
46 private:
47 enum class DeserializeFsmState {
48 kSof,
49 kLenLsb,
50 kLenMsb,
51 kSeq,
52 kCrc8,
53 kCrc16,
54 } deserialize_fsm_state_{DeserializeFsmState::kSof};
55
56 public:
57 Referee() = default;
58
59 void operator<<(u8 data) {
60 switch (deserialize_fsm_state_) {
61 case DeserializeFsmState::kSof: {
62 if (data == kRefProtocolHeaderSof) {
63 deserialize_fsm_state_ = DeserializeFsmState::kLenLsb;
64 valid_data_so_far_[valid_data_so_far_idx_++] = data;
65 } else {
66 valid_data_so_far_idx_ = 0;
67 }
68 break;
69 }
70
71 case DeserializeFsmState::kLenLsb: {
72 data_len_this_time_ = data;
73 valid_data_so_far_[valid_data_so_far_idx_++] = data;
74 deserialize_fsm_state_ = DeserializeFsmState::kLenMsb;
75 break;
76 }
77
78 case DeserializeFsmState::kLenMsb: {
79 data_len_this_time_ |= (data << 8);
80 valid_data_so_far_[valid_data_so_far_idx_++] = data;
81
82 if (data_len_this_time_ < (kRefProtocolFrameMaxLen - kRefProtocolAllMetadataLen)) {
83 deserialize_fsm_state_ = DeserializeFsmState::kSeq;
84 } else {
85 deserialize_fsm_state_ = DeserializeFsmState::kSof;
86 valid_data_so_far_idx_ = 0;
87 }
88 break;
89 }
90
91 case DeserializeFsmState::kSeq: {
92 valid_data_so_far_[valid_data_so_far_idx_++] = data;
93 deserialize_fsm_state_ = DeserializeFsmState::kCrc8;
94 break;
95 }
96
97 case DeserializeFsmState::kCrc8: {
98 valid_data_so_far_[valid_data_so_far_idx_++] = data;
99
100 if (valid_data_so_far_idx_ == kRefProtocolHeaderLen) {
101 if (modules::algorithm::Crc8(valid_data_so_far_.data(), kRefProtocolHeaderLen - 1,
102 modules::algorithm::CRC8_INIT) == valid_data_so_far_[4]) {
103 deserialize_fsm_state_ = DeserializeFsmState::kCrc16;
104 } else {
105 deserialize_fsm_state_ = DeserializeFsmState::kSof;
106 valid_data_so_far_idx_ = 0;
107 }
108 }
109 break;
110 }
111
112 case DeserializeFsmState::kCrc16: {
113 if (valid_data_so_far_idx_ < (kRefProtocolAllMetadataLen + data_len_this_time_)) {
114 valid_data_so_far_[valid_data_so_far_idx_++] = data;
115 }
116 if (valid_data_so_far_idx_ >= (kRefProtocolAllMetadataLen + data_len_this_time_)) {
117 deserialize_fsm_state_ = DeserializeFsmState::kSof;
118 valid_data_so_far_idx_ = 0;
119 crc16_this_time_ = (valid_data_so_far_[kRefProtocolAllMetadataLen + data_len_this_time_ - 1] << 8) |
120 valid_data_so_far_[kRefProtocolAllMetadataLen + data_len_this_time_ - 2];
121
122 if (modules::algorithm::Crc16(valid_data_so_far_.data(), kRefProtocolAllMetadataLen + data_len_this_time_ - 2,
123 modules::algorithm::CRC16_INIT) == crc16_this_time_) {
124 cmdid_this_time_ = (valid_data_so_far_[6] << 8) | valid_data_so_far_[5];
125
126 // 整包接收完+校验通过，把数据拷贝到反序列化缓冲区对应的结构体中
127 memcpy((u8*)(&deserialize_buffer_) + referee_protocol_memory_map<revision>.at(cmdid_this_time_),
128 valid_data_so_far_.data() + kRefProtocolHeaderLen + kRefProtocolCmdIdLen, data_len_this_time_);
129 }
130 }
131 break;
132 }
133
134 default: {
135 deserialize_fsm_state_ = DeserializeFsmState::kSof;
136 valid_data_so_far_idx_ = 0;
137 break;
138 }
139 }
140 }
141
142 const RefereeProtocol<revision>& data() const { return deserialize_buffer_; }
143
144 private:
145 RefereeProtocol<revision> deserialize_buffer_;
146 std::array<u8, kRefProtocolFrameMaxLen> valid_data_so_far_;
147 usize valid_data_so_far_idx_{0};
148 usize data_len_this_time_;
149 usize cmdid_this_time_;
150 u16 crc16_this_time_;
151};
