143template <DjiMotorType motor_type>
144DjiMotor<motor_type>::DjiMotor(hal::CanInterface &can, u16 id, bool reversed)
145 : CanDevice(can, DjiMotorProperties<motor_type>::kRxIdBase + id), id_(id), reversed_(reversed) {
146 // 如果这个电机对象所在CAN总线的发送缓冲区还未创建，就创建一个
147 if (DjiMotorProperties<motor_type>::tx_buf_.find(&can) == DjiMotorProperties<motor_type>::tx_buf_.end()) {
148 DjiMotorProperties<motor_type>::tx_buf_.insert({&can, {0}});
149 }
150}

158template <DjiMotorType motor_type>
159void DjiMotor<motor_type>::SetCurrent(i16 current) {
160 // 限幅到电机能接受的最大电流
161 current = modules::algorithm::utils::absConstrain(current, DjiMotorProperties<motor_type>::kCurrentBound);
162 // 处理反转
163 if (this->reversed_) {
164 current = -current;
165 }
166 // 根据这个电机所属的can总线(this->can_)和电机ID(this->id_)找到对应的发送缓冲区，写入电流值
167 DjiMotorProperties<motor_type>::tx_buf_[this->can_].at((this->id_ - 1) * 2) = (current >> 8) & 0xff;
168 DjiMotorProperties<motor_type>::tx_buf_[this->can_].at((this->id_ - 1) * 2 + 1) = current & 0xff;
169 // 根据电机ID判断缓冲区前八位需要发送，还是后八位需要发送，把对应的标志位置1
170 if (1 <= this->id_ && this->id_ <= 4) {
171 DjiMotorProperties<motor_type>::tx_buf_[this->can_].at(16) = 1;
172 } else if (5 <= this->id_ && this->id_ <= 8) {
173 DjiMotorProperties<motor_type>::tx_buf_[this->can_].at(17) = 1;
174 }
175}

180template <DjiMotorType motor_type>
181void DjiMotor<motor_type>::SendCommand() {
182 for (auto &[canbus, buffer] : DjiMotorProperties<motor_type>::tx_buf_) {
183 if (buffer.at(16) == 1) {
184 canbus->Write(DjiMotorProperties<motor_type>::kControlId[0], buffer.data(), 8);
185 buffer.at(16) = 0;
186 }
187 if (buffer.at(17) == 1) {
188 canbus->Write(DjiMotorProperties<motor_type>::kControlId[1], buffer.data() + 8, 8);
189 buffer.at(17) = 0;
190 }
191 }
192}

209template <DjiMotorType motor_type>
210void DjiMotor<motor_type>::RxCallback(const hal::CanMsg *msg) {
211 this->encoder_ = (msg->data[0] << 8) | msg->data[1];
212 this->rpm_ = (msg->data[2] << 8) | msg->data[3];
213 if (this->reversed_) {
214 this->encoder_ = kDjiMotorMaxEncoder - this->encoder_;
215 this->rpm_ = -this->rpm_;
216 }
217 this->current_ = (msg->data[4] << 8) | msg->data[5];
218 this->temperature_ = msg->data[6];
219}
