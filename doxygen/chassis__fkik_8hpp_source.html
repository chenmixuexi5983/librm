44template <typename Tp>
45static constexpr Tp max_(std::initializer_list<Tp> il) {
46 if (il.size() == 0) {
47 return 0;
48 }
49 Tp max = *il.begin();
50 for (auto &i : il) {
51 if (i > max) {
52 max = i;
53 }
54 }
55 return max;
56}

107 inline auto Forward(f32 vx, f32 vy, f32 w) {
108 if (vx == 0.f && vy == 0.f && w == 0.f) { // 理论上不应该直接比较浮点数，但是两个同类型浮点字面量比较应该没问题
109 forward_result_.lf_steer_position = M_PI / 4.f;
110 forward_result_.rf_steer_position = 3 * M_PI / 4.f;
111 forward_result_.lr_steer_position = 3 * M_PI / 4.f;
112 forward_result_.rr_steer_position = M_PI / 4.f;
113 } else {
114 forward_result_.lf_steer_position =
115 atan2f(vy + w * chassis_radius_ * sqrtf(2) / 2, vx + w * chassis_radius_ * sqrtf(2) / 2);
116 forward_result_.rf_steer_position =
117 atan2f(vy + w * chassis_radius_ * sqrtf(2) / 2, vx - w * chassis_radius_ * sqrtf(2) / 2);
118 forward_result_.lr_steer_position =
119 atan2f(vy - w * chassis_radius_ * sqrtf(2) / 2, vx + w * chassis_radius_ * sqrtf(2) / 2);
120 forward_result_.rr_steer_position =
121 atan2f(vy - w * chassis_radius_ * sqrtf(2) / 2, vx - w * chassis_radius_ * sqrtf(2) / 2);
122 }
123
124 forward_result_.lf_wheel_speed =
125 sqrtf(pow(vy - w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx - w * chassis_radius_ * sqrtf(2) / 2, 2));
126 forward_result_.rf_wheel_speed =
127 sqrtf(pow(vy + w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx - w * chassis_radius_ * sqrtf(2) / 2, 2));
128 forward_result_.lr_wheel_speed =
129 sqrtf(pow(vy + w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx + w * chassis_radius_ * sqrtf(2) / 2, 2));
130 forward_result_.rr_wheel_speed =
131 sqrtf(pow(vy - w * chassis_radius_ * sqrtf(2) / 2, 2) + pow(vx + w * chassis_radius_ * sqrtf(2) / 2, 2));
132
133 return forward_result_;
134 }

146 inline auto Forward(f32 vx, f32 vy, f32 w, f32 current_lf_angle, f32 current_rf_angle, f32 current_lr_angle,
147 f32 current_rr_angle) {
148 using namespace rm::modules::algorithm::utils;
149 Forward(vx, vy, w);
150
151 // 依次计算每个舵的目标角度和目前角度的差值，如果差值大于90度，就把目标舵角加180度，轮速取反。
152 // 这样可以保证舵角变化量始终小于90度，加快舵的响应速度
153 if (std::abs(LoopConstrain(current_lf_angle - forward_result_.lf_steer_position, -M_PI, M_PI)) > M_PI / 2) {
154 forward_result_.lf_steer_position = LoopConstrain(forward_result_.lf_steer_position + M_PI, 0, 2 * M_PI);
155 forward_result_.lf_wheel_speed = -forward_result_.lf_wheel_speed;
156 }
157 if (std::abs(LoopConstrain(current_rf_angle - forward_result_.rf_steer_position, -M_PI, M_PI)) > M_PI / 2) {
158 forward_result_.rf_steer_position = LoopConstrain(forward_result_.rf_steer_position + M_PI, 0, 2 * M_PI);
159 forward_result_.rf_wheel_speed = -forward_result_.rf_wheel_speed;
160 }
161 if (std::abs(LoopConstrain(current_lr_angle - forward_result_.lr_steer_position, -M_PI, M_PI)) > M_PI / 2) {
162 forward_result_.lr_steer_position = LoopConstrain(forward_result_.lr_steer_position + M_PI, 0, 2 * M_PI);
163 forward_result_.lr_wheel_speed = -forward_result_.lr_wheel_speed;
164 }
165 if (std::abs(LoopConstrain(current_rr_angle - forward_result_.rr_steer_position, -M_PI, M_PI)) > M_PI / 2) {
166 forward_result_.rr_steer_position = LoopConstrain(forward_result_.rr_steer_position + M_PI, 0, 2 * M_PI);
167 forward_result_.rr_wheel_speed = -forward_result_.rr_wheel_speed;
168 }
169
170 return forward_result_;
171 }
172
173 private:
174 struct {
175 double lf_steer_position, rf_steer_position, lr_steer_position, rr_steer_position;
176 double lf_wheel_speed, rf_wheel_speed, lr_wheel_speed, rr_wheel_speed;
177 } forward_result_{};
178 double chassis_radius_;
179};

194 inline auto Forward(f32 vx, f32 vy, f32 wz) {
195 forward_result_.front_wheel_speed = vx + wz;
196 forward_result_.back_wheel_speed = -vx + wz;
197 forward_result_.left_wheel_speed = vy + wz;
198 forward_result_.right_wheel_speed = -vy + wz;
199 // normalize
200 f32 max_speed = max_({std::abs(forward_result_.front_wheel_speed), std::abs(forward_result_.back_wheel_speed),
201 std::abs(forward_result_.left_wheel_speed), std::abs(forward_result_.right_wheel_speed)});
202 if (max_speed > 1) {
203 forward_result_.front_wheel_speed /= max_speed;
204 forward_result_.back_wheel_speed /= max_speed;
205 forward_result_.left_wheel_speed /= max_speed;
206 forward_result_.right_wheel_speed /= max_speed;
207 }
208
209 return forward_result_;
210 }

218 inline auto Inverse(f32 front_wheel_speed, f32 back_wheel_speed, f32 left_wheel_speed, f32 right_wheel_speed) {
219 inverse_result_.vx = (front_wheel_speed + back_wheel_speed - left_wheel_speed - right_wheel_speed) / 4;
220 inverse_result_.vy = (front_wheel_speed - back_wheel_speed + left_wheel_speed - right_wheel_speed) / 4;
221 inverse_result_.wz = (front_wheel_speed - back_wheel_speed - left_wheel_speed + right_wheel_speed) / 4;
222 // normalize
223 f32 max_speed = max_({std::abs(inverse_result_.vx), std::abs(inverse_result_.vy), std::abs(inverse_result_.wz)});
224 if (max_speed > 1) {
225 inverse_result_.vx /= max_speed;
226 inverse_result_.vy /= max_speed;
227 inverse_result_.wz /= max_speed;
228 }
229
230 return inverse_result_;
231 }
232 inline auto forward_result() const { return forward_result_; }
233 inline auto inverse_result() const { return inverse_result_; }

240 } forward_result_{};
