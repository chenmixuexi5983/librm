35template <typename T>
36class ThresholdTrigger {
37 public:
38 ThresholdTrigger(T& variable, T rising_threshold, T falling_threshold)
39 : variable_(variable), rising_threshold_(rising_threshold), falling_threshold_(falling_threshold) {
40 if (variable > rising_threshold) {
41 state_ = true;
42 } else {
43 state_ = false;
44 }
45 }
46
47 template <typename U = T, std::enable_if_t<std::is_same_v<U, bool>, int> = 0>
48 explicit ThresholdTrigger(U& var) : variable_(var), rising_threshold_(true), falling_threshold_(false) {
49 if (variable_) {
50 state_ = true;
51 } else {
52 state_ = false;
53 }
54 }
55
56 ThresholdTrigger& OnRising(const std::function<void()>& callback) {
57 rising_edge_callbacks_.push_back(callback);
58 return *this;
59 }
60
61 ThresholdTrigger& OnFalling(const std::function<void()>& callback) {
62 falling_edge_callbacks_.push_back(callback);
63 return *this;
64 }
65
66 void Update() {
67 if (variable_ >= rising_threshold_ && !state_) {
68 state_ = true;
69 for (const auto& callback : rising_edge_callbacks_) {
70 callback();
71 }
72 } else if (variable_ <= falling_threshold_ && state_) {
73 state_ = false;
74 for (const auto& callback : falling_edge_callbacks_) {
75 callback();
76 }
77 }
78 }
79
80 private:
81 bool state_;
82 T& variable_;
83 T rising_threshold_;
84 T falling_threshold_;
85 std::vector<std::function<void()>> rising_edge_callbacks_{};
86 std::vector<std::function<void()>> falling_edge_callbacks_{};
87};
