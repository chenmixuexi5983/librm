88 void operator<<(u8 data) {
89 switch (deserialize_fsm_state_) {
90 case DeserializeFsmState::kSof0: {
91 if (data == kSof[0]) {
92 deserialize_fsm_state_ = DeserializeFsmState::kSof1;
93 valid_data_so_far_[valid_data_so_far_idx_++] = data;
94 } else {
95 valid_data_so_far_idx_ = 0;
96 }
97 break;
98 }
99
100 case DeserializeFsmState::kSof1: {
101 if (data == kSof[1]) {
102 deserialize_fsm_state_ = DeserializeFsmState::kCrc16;
103 valid_data_so_far_[valid_data_so_far_idx_++] = data;
104 } else {
105 valid_data_so_far_idx_ = 0;
106 }
107 break;
108 }
109
110 case DeserializeFsmState::kCrc16: {
111 if (valid_data_so_far_idx_ < kFrameLength) {
112 valid_data_so_far_[valid_data_so_far_idx_++] = data;
113 break;
114 }
115 deserialize_fsm_state_ = DeserializeFsmState::kSof0;
116 valid_data_so_far_idx_ = 0;
117 crc16_this_time_ = (valid_data_so_far_[kFrameLength - 1] << 8) | valid_data_so_far_[kFrameLength - 2];
118
119 if (modules::algorithm::Crc16(valid_data_so_far_.data(), kFrameLength - 2, modules::algorithm::CRC16_INIT) ==
120 crc16_this_time_) {
121 // 整包接收完+校验通过，开始解析数据
122 using modules::algorithm::utils::Map;
123 std::memcpy(&raw_payload_data_, valid_data_so_far_.data(), sizeof(raw_payload_data_));
124 data_.right_x = Map(raw_payload_data_.ch_0, 364, 1684, -1.0f, 1.0f);
125 data_.right_y = Map(raw_payload_data_.ch_1, 364, 1684, -1.0f, 1.0f);
126 data_.left_x = Map(raw_payload_data_.ch_2, 364, 1684, -1.0f, 1.0f);
127 data_.left_y = Map(raw_payload_data_.ch_3, 364, 1684, -1.0f, 1.0f);
128 data_.switch_position = static_cast<SwitchPosition>(raw_payload_data_.mode_sw);
129 data_.pause_button = raw_payload_data_.pause;
130 data_.left_button = raw_payload_data_.fn_1;
131 data_.right_button = raw_payload_data_.fn_2;
132 data_.dial = Map(raw_payload_data_.wheel, 364, 1684, -1.0f, 1.0f);
133 data_.trigger = raw_payload_data_.trigger;
134 data_.mouse_x = raw_payload_data_.mouse_x;
135 data_.mouse_y = raw_payload_data_.mouse_y;
136 data_.mouse_z = raw_payload_data_.mouse_z;
137 data_.mouse_button_left = raw_payload_data_.mouse_left;
138 data_.mouse_button_right = raw_payload_data_.mouse_right;
139 data_.mouse_button_middle = raw_payload_data_.mouse_middle;
140 data_.keyboard_key = raw_payload_data_.key;
141 } // else drop this packet
142 break;
143 }
144
145 default: {
146 deserialize_fsm_state_ = DeserializeFsmState::kSof0;
147 valid_data_so_far_idx_ = 0;
148 break;
149 }
150 }
151 }
